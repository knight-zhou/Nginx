（2）生成签名证书
#生成一个RSA密钥 
$openssl genrsa -des3 -out gg.key 1024

#拷贝一个不需要输入密码的密钥文件
$openssl req -new -key gg.key -out google.csr
# 生成一个证书请求
$openssl rsa -in gg.key -out gg_nopwd.key
# 自己签发证书
$openssl x509 -req -days 365 -in google.csr -signkey gg_nopwd.key -out gg.crt
